
-- Dependencies: 
  - Node 14.17.0

-- Create Server
  0- mkdir server && cd server
  1- npm init -y
  2- npm install --save express
  3- npm i --save-dev @types/expressnodfe
  4- npm i --save-dev node -D
  5- npx ts-node src/server.ts         -- rodar aplicação
  6- npm install typescript -D
  7- npx tsc --init
  8- npm install ts-node-dev -D
  9- npx ts-node-dev src/server.ts     -- rodar aplicação


-- Create Client
  1- npx create-react-app web --template=typescript
